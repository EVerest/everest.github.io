775 // Generates async sending callbacks
776 template <class RequestType, class ResponseType>
777 std::function<ResponseType(RequestType)> send_callback(MessageType expected_response_message_type) {
778 return [this, expected_response_message_type](auto request) {
779 const auto enhanced_response =
780 this->message_dispatcher->dispatch_call_async(ocpp::Call<RequestType>(request)).get();
781 if (enhanced_response.messageType != expected_response_message_type) {
782 throw UnexpectedMessageTypeFromCSMS(
783 std::string("Got unexpected message type from CSMS, expected: ") +
784 conversions::messagetype_to_string(expected_response_message_type) +
785 ", got: " + conversions::messagetype_to_string(enhanced_response.messageType));
786 }
787 ocpp::CallResult<ResponseType> call_result = enhanced_response.message;
788 return call_result.msg;
789 };
790 }
