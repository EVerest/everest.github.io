45 /// \brief Describes the individual state of a single connector
46 struct FullConnectorStatus {
47 /// \brief Operative/Inoperative status, usually set by the CSMS
48 OperationalStatusEnum individual_operational_status;
49 /// \brief True if the connector has an active (uncleared) error, assumed false on boot
50 bool faulted;
51 /// \brief True if the connector has an active reservation, assumed false on boot

53 /// \brief True if the connector has a cable plugged in, assumed false on boot
